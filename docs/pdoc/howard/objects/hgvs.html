HGVS language currently implemented.

HGVS = ALLELE | PREFIX_NAME : ALLELE

PREFIX_NAME = TRANSCRIPT | TRANSCRIPT '(' GENE ')'

TRANSCRIPT = TRANSCRIPT_NAME | TRANSCRIPT_NAME '.' TRANSCRIPT_VERSION

TRANSCRIPT_VERSION = NUMBER

ALLELE = 'c.' CDNA_ALLELE # cDNA | 'g.' GENOMIC_ALLELE # genomic | 'm.' MIT_ALLELE # mitochondrial sequence | 'n.' NC_ALLELE # non-coding RNA reference sequence | 'r.' RNA_ALLELE # RNA sequence (like r.76a>u) | 'p.' PROTEIN_ALLELE # protein sequence (like p.Lys76Asn)

NC_ALLELE = RNA_ALLELE = CDNA_ALLELE = CDNA_COORD SINGLE_BASE_CHANGE | CDNA_COORD_RANGE MULTI_BASE_CHANGE

GENOMIC_ALLELE = MIT_ALLELE = COORD SINGLE_BASE_CHANGE | COORD_RANGE MULTI_BASE_CHANGE

SINGLE_BASE_CHANGE = CDNA_ALLELE = CDNA_COORD BASE '=' # no change | CDNA_COORD BASE '>' BASE # substitution | CDNA_COORD 'ins' BASE # 1bp insertion | CDNA_COORD 'del' BASE # 1bp deletion | CDNA_COORD 'dup' BASE # 1bp duplication | CDNA_COORD 'ins' # 1bp insertion | CDNA_COORD 'del' # 1bp deletion | CDNA_COORD 'dup' # 1bp duplication | CDNA_COORD 'del' BASE 'ins' BASE # 1bp indel | CDNA_COORD 'delins' BASE # 1bp indel

 302# The HGVSRegex class is used for working with regular expressions related to the Human Genome
 303# Variation Society (HGVS) nomenclature.
 304class HGVSRegex(object):
 305    """
 306    All regular expression for HGVS names.
 307    """
 308
 309    # DNA syntax
 310    # http://www.hgvs.org/mutnomen/standards.html#nucleotide
 311    BASE = r"[acgtbdhkmnrsvwyACGTBDHKMNRSVWY]|\d+"
 312    BASES = r"[acgtbdhkmnrsvwyACGTBDHKMNRSVWY]+|\d+"
 313    DNA_REF = "(?P<ref>" + BASES + ")"
 314    DNA_ALT = "(?P<alt>" + BASES + ")"
 315
 316    # Mutation types
 317    EQUAL = "(?P<mutation_type>=)"
 318    SUB = "(?P<mutation_type>>)"
 319    INS = "(?P<mutation_type>ins)"
 320    DEL = "(?P<mutation_type>del)"
 321    DUP = "(?P<mutation_type>dup)"
 322
 323    # Simple coordinate syntax
 324    COORD_START = r"(?P<start>\d+)"
 325    COORD_END = r"(?P<end>\d+)"
 326    COORD_RANGE = COORD_START + "_" + COORD_END
 327
 328    # cDNA coordinate syntax
 329    CDNA_COORD = (
 330        r"(?P<coord_prefix>|-|\*)(?P<coord>\d+)"
 331        r"((?P<offset_prefix>-|\+)(?P<offset>\d+))?"
 332    )
 333    CDNA_START = (
 334        r"(?P<start>(?P<start_coord_prefix>|-|\*)(?P<start_coord>\d+)"
 335        r"((?P<start_offset_prefix>-|\+)(?P<start_offset>\d+))?)"
 336    )
 337    CDNA_END = (
 338        r"(?P<end>(?P<end_coord_prefix>|-|\*)(?P<end_coord>\d+)"
 339        r"((?P<end_offset_prefix>-|\+)(?P<end_offset>\d+))?)"
 340    )
 341    CDNA_RANGE = CDNA_START + "_" + CDNA_END
 342
 343    # cDNA allele syntax
 344    CDNA_ALLELE = [
 345        # No change
 346        CDNA_START + EQUAL,
 347        CDNA_START + DNA_REF + EQUAL,
 348        # Substitution
 349        CDNA_START + DNA_REF + SUB + DNA_ALT,
 350        # 1bp insertion, deletion, duplication
 351        CDNA_START + INS + DNA_ALT,
 352        CDNA_START + DEL + DNA_REF,
 353        CDNA_START + DUP + DNA_REF,
 354        CDNA_START + DEL,
 355        CDNA_START + DUP,
 356        # Insertion, deletion, duplication
 357        COORD_RANGE + EQUAL,
 358        CDNA_RANGE + INS + DNA_ALT,
 359        CDNA_RANGE + DEL + DNA_REF,
 360        CDNA_RANGE + DUP + DNA_REF,
 361        CDNA_RANGE + DEL,
 362        CDNA_RANGE + DUP,
 363        # Indels
 364        "(?P<delins>" + CDNA_START + "del" + DNA_REF + "ins" + DNA_ALT + ")",
 365        "(?P<delins>" + CDNA_RANGE + "del" + DNA_REF + "ins" + DNA_ALT + ")",
 366        "(?P<delins>" + CDNA_START + "delins" + DNA_ALT + ")",
 367        "(?P<delins>" + CDNA_RANGE + "delins" + DNA_ALT + ")",
 368    ]
 369
 370    CDNA_ALLELE_REGEXES = [re.compile("^" + regex + "$") for regex in CDNA_ALLELE]
 371
 372    # Peptide syntax
 373    PEP = "([A-Z]([a-z]{2}))+"
 374    PEP_REF = "(?P<ref>" + PEP + ")"
 375    PEP_REF2 = "(?P<ref2>" + PEP + ")"
 376    PEP_ALT = "(?P<alt>" + PEP + ")"
 377
 378    PEP_EXTRA = r"(?P<extra>(|=|\?)(|fs))"
 379
 380    # Peptide allele syntax
 381    PEP_ALLELE = [
 382        # No peptide change
 383        # Example: Glu1161=
 384        PEP_REF + COORD_START + PEP_EXTRA,
 385        # Peptide change
 386        # Example: Glu1161Ser
 387        PEP_REF + COORD_START + PEP_ALT + PEP_EXTRA,
 388        # Peptide indel
 389        # Example: Glu1161_Ser1164?fs
 390        "(?P<delins>"
 391        + PEP_REF
 392        + COORD_START
 393        + "_"
 394        + PEP_REF2
 395        + COORD_END
 396        + PEP_EXTRA
 397        + ")",
 398        "(?P<delins>"
 399        + PEP_REF
 400        + COORD_START
 401        + "_"
 402        + PEP_REF2
 403        + COORD_END
 404        + PEP_ALT
 405        + PEP_EXTRA
 406        + ")",
 407    ]
 408
 409    PEP_ALLELE_REGEXES = [re.compile("^" + regex + "$") for regex in PEP_ALLELE]
 410
 411    # Genomic allele syntax
 412    GENOMIC_ALLELE = [
 413        # No change
 414        COORD_START + EQUAL,
 415        COORD_START + DNA_REF + EQUAL,
 416        # Substitution
 417        COORD_START + DNA_REF + SUB + DNA_ALT,
 418        # 1bp insertion, deletion, duplication
 419        COORD_START + INS + DNA_ALT,
 420        COORD_START + DEL + DNA_REF,
 421        COORD_START + DUP + DNA_REF,
 422        COORD_START + DEL,
 423        COORD_START + DUP,
 424        # Insertion, deletion, duplication
 425        COORD_RANGE + EQUAL,
 426        COORD_RANGE + INS + DNA_ALT,
 427        COORD_RANGE + DEL + DNA_REF,
 428        COORD_RANGE + DUP + DNA_REF,
 429        COORD_RANGE + DEL,
 430        COORD_RANGE + DUP,
 431        # Indels
 432        "(?P<delins>" + COORD_START + "del" + DNA_REF + "ins" + DNA_ALT + ")",
 433        "(?P<delins>" + COORD_RANGE + "del" + DNA_REF + "ins" + DNA_ALT + ")",
 434        "(?P<delins>" + COORD_START + "delins" + DNA_ALT + ")",
 435        "(?P<delins>" + COORD_RANGE + "delins" + DNA_ALT + ")",
 436    ]
 437
 438    GENOMIC_ALLELE_REGEXES = [re.compile("^" + regex + "$") for regex in GENOMIC_ALLELE]

 481# The InvalidHGVSName class is a subclass of ValueError and is used to represent an error when an
 482# invalid HGVS name is encountered.
 483class InvalidHGVSName(ValueError):
 484
 485    def __init__(self, name: str = "", part: str = "name", reason: str = "") -> None:
 486        """
 487        The function initializes an InvalidHGVSName object with a message, name, part, and reason.
 488
 489        :param name: The name parameter is a string that represents the invalid HGVS name. It is the
 490        name that is considered invalid and does not meet the required criteria
 491        :type name: str
 492        :param part: The "part" parameter represents the part of the HGVS (Human Genome Variation
 493        Society) name that is invalid. It is used to provide more specific information about the error
 494        that occurred, defaults to name
 495        :type part: str (optional)
 496        :param reason: The "reason" parameter is an optional argument that provides additional
 497        information or context for why the HGVS name is considered invalid. It can be used to provide
 498        specific details about the error or to explain why the name does not meet the required criteria
 499        :type reason: str
 500        """

 660    def parse_prefix(self, prefix: str):
 661        """
 662        The `parse_prefix` function is used to parse a HGVS prefix (gene/transcript/chromosome) and
 663        assign the parsed values to the corresponding attributes of the object.
 664
 665        :param prefix: The `prefix` parameter is a string that represents a HGVS prefix, which can be a
 666        gene, transcript, or chromosome identifier. It is used to determine the type of prefix and
 667        assign the parsed values to the corresponding attributes of the object
 668        :type prefix: str
 669        :return: The function `parse_prefix` returns the parsed values for the transcript and gene
 670        attributes, or sets the chrom or gene attributes based on the given prefix.
 671        """
 672
 673        self.prefix = prefix
 674
 675        # No prefix.
 676        if prefix == "":
 677            self.chrom = ""
 678            self.transcript = ""
 679            self.gene = ""
 680            return
 681
 682        # Transcript and gene given with parens.
 683        # example: NM_007294.3(BRCA1):c.2207A>C
 684        match = re.match(r"^(?P<transcript>[^(]+)\((?P<gene>[^)]+)\)$", prefix)
 685        if match:
 686            self.transcript = match.group("transcript")
 687            self.gene = match.group("gene")
 688            return
 689
 690        # Transcript and gene given with braces.
 691        # example: BRCA1{NM_007294.3}:c.2207A>C
 692        match = re.match(r"^(?P<gene>[^{]+){(?P<transcript>[^}]+)}$", prefix)
 693        if match:
 694            self.transcript = match.group("transcript")
 695            self.gene = match.group("gene")
 696            return
 697
 698        # Determine using Ensembl type.
 699        if prefix.startswith("ENST"):
 700            self.transcript = prefix
 701            return
 702
 703        # Determine using LRG type.
 704        if prefix.startswith("LRG_"):
 705            self.transcript = prefix
 706            return
 707
 708        # Determine using refseq type.
 709        refseq_type = get_refseq_type(prefix)
 710        if refseq_type in ("mRNA", "RNA"):
 711            self.transcript = prefix
 712            return
 713
 714        # Determine using refseq type.
 715        if prefix.startswith(CHROM_PREFIX) or refseq_type == "genomic":
 716            self.chrom = prefix
 717            return
 718
 719        # Assume gene name.
 720        self.gene = prefix

 722    def parse_allele(self, allele: str) -> None:
 723        """
 724        The function `parse_allele` parses a HGVS allele description and determines the kind of HGVS
 725        name (c., p., g., etc.) and the mutation type.
 726
 727        Some examples include:
 728          cDNA substitution: c.101A>C,
 729          cDNA indel: c.3428delCinsTA, c.1000_1003delATG, c.1000_1001insATG
 730          No protein change: p.Glu1161=
 731          Protein change: p.Glu1161Ser
 732          Protein frameshift: p.Glu1161_Ser1164?fs
 733          Genomic substitution: g.1000100A>T
 734          Genomic indel: g.1000100_1000102delATG
 735
 736        :param allele: The `allele` parameter is a string that represents a HGVS allele description. It
 737        can contain various types of mutations, such as cDNA substitutions, cDNA indels, protein
 738        changes, protein frameshifts, genomic substitutions, and genomic indels. The purpose of the
 739        `parse_allele`
 740        :type allele: str
 741        """
 742
 743        if "." not in allele:
 744            raise InvalidHGVSName(
 745                allele, "allele", 'expected kind "c.", "p.", "g.", etc'
 746            )
 747
 748        # Determine HGVS name kind.
 749        kind, details = allele.split(".", 1)
 750        self.kind = kind
 751        self.mutation_type = None
 752
 753        if kind in ("c", "n"):
 754            self.parse_cdna(details)
 755            if kind == "n":  # Ensure no 3'UTR or 5'UTR coords in non-coding
 756                if self.cdna_start.coord < 0:
 757                    raise InvalidHGVSName(
 758                        allele,
 759                        "allele",
 760                        "Non-coding transcript cannot contain negative (5'UTR) coordinates",
 761                    )
 762                if (
 763                    self.cdna_start.landmark == "cdna_stop"
 764                    or self.cdna_end
 765                    and self.cdna_end.landmark == "cdna_stop"
 766                ):
 767                    raise InvalidHGVSName(
 768                        allele,
 769                        "allele",
 770                        "Non-coding transcript cannot contain '*' (3'UTR) coordinates",
 771                    )
 772        elif kind == "p":
 773            self.parse_protein(details)
 774        elif kind in ("g", "m"):
 775            self.parse_genome(details)
 776        else:
 777            raise NotImplementedError("unknown kind: %s" % allele)

 779    def parse_cdna(self, details: str) -> None:
 780        """
 781        The function `parse_cdna` is used to parse a HGVS cDNA name and extract information such as
 782        mutation type, coordinates, and alleles.
 783
 784        Some examples include:
 785          Substitution: 101A>C,
 786          Indel: 3428delCinsTA, 1000_1003delATG, 1000_1001insATG
 787
 788        :param details: The `details` parameter is a string that represents a HGVS cDNA name. It
 789        contains information about a genetic mutation, such as a substitution or an indel, along with
 790        the specific coordinates and alleles involved in the mutation
 791        :type details: str
 792        :return: None.
 793        """
 794
 795        for regex in HGVSRegex.CDNA_ALLELE_REGEXES:
 796            match = re.match(regex, details)
 797            if match:
 798                groups = match.groupdict()
 799
 800                # Parse mutation type.
 801                if groups.get("delins"):
 802                    self.mutation_type = "delins"
 803                else:
 804                    self.mutation_type = groups["mutation_type"]
 805
 806                # Parse coordinates.
 807                self.cdna_start = CDNACoord(string=groups.get("start"))
 808                if groups.get("end"):
 809                    self.cdna_end = CDNACoord(string=groups.get("end"))
 810                else:
 811                    self.cdna_end = CDNACoord(string=groups.get("start"))
 812
 813                # Parse alleles.
 814                self.ref_allele = groups.get("ref", "")
 815                self.alt_allele = groups.get("alt", "")
 816
 817                # Convert numerical allelles.
 818                if self.ref_allele.isdigit():
 819                    self.ref_allele = "N" * int(self.ref_allele)
 820                if self.alt_allele.isdigit():
 821                    self.alt_allele = "N" * int(self.alt_allele)
 822
 823                # Convert duplication alleles.
 824                if self.mutation_type == "dup":
 825                    self.alt_allele = self.ref_allele * 2
 826
 827                # Convert no match alleles.
 828                if self.mutation_type == "=":
 829                    self.alt_allele = self.ref_allele
 830                return
 831
 832        raise InvalidHGVSName(details, "cDNA allele")

 661    def parse_prefix(self, prefix: str):
 662        """
 663        The `parse_prefix` function is used to parse a HGVS prefix (gene/transcript/chromosome) and
 664        assign the parsed values to the corresponding attributes of the object.
 665
 666        :param prefix: The `prefix` parameter is a string that represents a HGVS prefix, which can be a
 667        gene, transcript, or chromosome identifier. It is used to determine the type of prefix and
 668        assign the parsed values to the corresponding attributes of the object
 669        :type prefix: str
 670        :return: The function `parse_prefix` returns the parsed values for the transcript and gene
 671        attributes, or sets the chrom or gene attributes based on the given prefix.
 672        """
 673
 674        self.prefix = prefix
 675
 676        # No prefix.
 677        if prefix == "":
 678            self.chrom = ""
 679            self.transcript = ""
 680            self.gene = ""
 681            return
 682
 683        # Transcript and gene given with parens.
 684        # example: NM_007294.3(BRCA1):c.2207A>C
 685        match = re.match(r"^(?P<transcript>[^(]+)\((?P<gene>[^)]+)\)$", prefix)
 686        if match:
 687            self.transcript = match.group("transcript")
 688            self.gene = match.group("gene")
 689            return
 690
 691        # Transcript and gene given with braces.
 692        # example: BRCA1{NM_007294.3}:c.2207A>C
 693        match = re.match(r"^(?P<gene>[^{]+){(?P<transcript>[^}]+)}$", prefix)
 694        if match:
 695            self.transcript = match.group("transcript")
 696            self.gene = match.group("gene")
 697            return
 698
 699        # Determine using Ensembl type.
 700        if prefix.startswith("ENST"):
 701            self.transcript = prefix
 702            return
 703
 704        # Determine using LRG type.
 705        if prefix.startswith("LRG_"):
 706            self.transcript = prefix
 707            return
 708
 709        # Determine using refseq type.
 710        refseq_type = get_refseq_type(prefix)
 711        if refseq_type in ("mRNA", "RNA"):
 712            self.transcript = prefix
 713            return
 714
 715        # Determine using refseq type.
 716        if prefix.startswith(CHROM_PREFIX) or refseq_type == "genomic":
 717            self.chrom = prefix
 718            return
 719
 720        # Assume gene name.
 721        self.gene = prefix

 723    def parse_allele(self, allele: str) -> None:
 724        """
 725        The function `parse_allele` parses a HGVS allele description and determines the kind of HGVS
 726        name (c., p., g., etc.) and the mutation type.
 727
 728        Some examples include:
 729          cDNA substitution: c.101A>C,
 730          cDNA indel: c.3428delCinsTA, c.1000_1003delATG, c.1000_1001insATG
 731          No protein change: p.Glu1161=
 732          Protein change: p.Glu1161Ser
 733          Protein frameshift: p.Glu1161_Ser1164?fs
 734          Genomic substitution: g.1000100A>T
 735          Genomic indel: g.1000100_1000102delATG
 736
 737        :param allele: The `allele` parameter is a string that represents a HGVS allele description. It
 738        can contain various types of mutations, such as cDNA substitutions, cDNA indels, protein
 739        changes, protein frameshifts, genomic substitutions, and genomic indels. The purpose of the
 740        `parse_allele`
 741        :type allele: str
 742        """
 743
 744        if "." not in allele:
 745            raise InvalidHGVSName(
 746                allele, "allele", 'expected kind "c.", "p.", "g.", etc'
 747            )
 748
 749        # Determine HGVS name kind.
 750        kind, details = allele.split(".", 1)
 751        self.kind = kind
 752        self.mutation_type = None
 753
 754        if kind in ("c", "n"):
 755            self.parse_cdna(details)
 756            if kind == "n":  # Ensure no 3'UTR or 5'UTR coords in non-coding
 757                if self.cdna_start.coord < 0:
 758                    raise InvalidHGVSName(
 759                        allele,
 760                        "allele",
 761                        "Non-coding transcript cannot contain negative (5'UTR) coordinates",
 762                    )
 763                if (
 764                    self.cdna_start.landmark == "cdna_stop"
 765                    or self.cdna_end
 766                    and self.cdna_end.landmark == "cdna_stop"
 767                ):
 768                    raise InvalidHGVSName(
 769                        allele,
 770                        "allele",
 771                        "Non-coding transcript cannot contain '*' (3'UTR) coordinates",
 772                    )
 773        elif kind == "p":
 774            self.parse_protein(details)
 775        elif kind in ("g", "m"):
 776            self.parse_genome(details)
 777        else:
 778            raise NotImplementedError("unknown kind: %s" % allele)

 780    def parse_cdna(self, details: str) -> None:
 781        """
 782        The function `parse_cdna` is used to parse a HGVS cDNA name and extract information such as
 783        mutation type, coordinates, and alleles.
 784
 785        Some examples include:
 786          Substitution: 101A>C,
 787          Indel: 3428delCinsTA, 1000_1003delATG, 1000_1001insATG
 788
 789        :param details: The `details` parameter is a string that represents a HGVS cDNA name. It
 790        contains information about a genetic mutation, such as a substitution or an indel, along with
 791        the specific coordinates and alleles involved in the mutation
 792        :type details: str
 793        :return: None.
 794        """
 795
 796        for regex in HGVSRegex.CDNA_ALLELE_REGEXES:
 797            match = re.match(regex, details)
 798            if match:
 799                groups = match.groupdict()
 800
 801                # Parse mutation type.
 802                if groups.get("delins"):
 803                    self.mutation_type = "delins"
 804                else:
 805                    self.mutation_type = groups["mutation_type"]
 806
 807                # Parse coordinates.
 808                self.cdna_start = CDNACoord(string=groups.get("start"))
 809                if groups.get("end"):
 810                    self.cdna_end = CDNACoord(string=groups.get("end"))
 811                else:
 812                    self.cdna_end = CDNACoord(string=groups.get("start"))
 813
 814                # Parse alleles.
 815                self.ref_allele = groups.get("ref", "")
 816                self.alt_allele = groups.get("alt", "")
 817
 818                # Convert numerical allelles.
 819                if self.ref_allele.isdigit():
 820                    self.ref_allele = "N" * int(self.ref_allele)
 821                if self.alt_allele.isdigit():
 822                    self.alt_allele = "N" * int(self.alt_allele)
 823
 824                # Convert duplication alleles.
 825                if self.mutation_type == "dup":
 826                    self.alt_allele = self.ref_allele * 2
 827
 828                # Convert no match alleles.
 829                if self.mutation_type == "=":
 830                    self.alt_allele = self.ref_allele
 831                return
 832
 833        raise InvalidHGVSName(details, "cDNA allele")

 836        """
 837        The function `parse_protein` is used to parse a HGVS protein name and extract information such
 838        as mutation type, coordinates, alleles, and additional details.

 841          No change: Glu1161=

 862                    self.mutation_type = ">"
 863

 865                self.start = int(groups.get("start"))
 866                if groups.get("end"):
 867                    self.end = int(groups.get("end"))
 868                else:
 869                    self.end = self.start

 881                self.pep_extra = groups.get("extra")
